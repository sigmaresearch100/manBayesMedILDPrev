\begin{tikzpicture}[

		node distance = 15mm and 15mm,
		manifest/.style = {
			rectangle,
			draw,
			thick,
			inner sep = 0pt,
			minimum width = 15mm,
			minimum height = 10mm
		},
		inv/.style = {

		},

			ellipse,
			draw,
			thick,

			align = center
		},
		mean/.style={
			regular polygon,
			regular polygon sides = 3,
			draw,
			thick,
			inner sep = 0pt,

			minimum height = 7mm
		},
		path/.style = {
			->,
			thick,
			>=stealth'
		},
		cov/.style = {
			<->,
			thick,

	\node[manifest] (X) [right = of inv] {$X$};
	\node[manifest] (M) [above left = of X] {$M$};

	\draw[path] (1) to node[below left] {$\delta_{M}$} (M);
	\draw[path] (1) to node {$\delta_{Y}$} (Y);
	\draw[cov] (X) to[out = 70, in = 110, looseness = 5] node[above] {$\sigma_{X}^{2}$} (X);
	\draw[cov] (varepsilonM) to[out = -290, in = -250, looseness = 5] node[above] {$\sigma^{2}_{\varepsilon_{M}}$} (varepsilonM);
	\draw[cov] (varepsilonY) to[out = -290, in = -250, looseness = 5] node[above] {$\sigma^{2}_{\varepsilon_{Y}}$} (varepsilonY);
	\draw[cov] (1) to[out = -60, in = -120, looseness = 5] node[below] {$1$} (1);
\end{tikzpicture}
